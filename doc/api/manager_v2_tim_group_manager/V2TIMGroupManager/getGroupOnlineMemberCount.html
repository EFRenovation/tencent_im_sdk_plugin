Future<V2TimValueCallback<int>> getGroupOnlineMemberCount(
    {@required String groupID}) async {
  return V2TimValueCallback<int>.fromJson(
    formatJson(
      await _channel.invokeMethod(
        "getGroupOnlineMemberCount",
        buildParam(
          {"groupID": groupID},
        ),
      ),
    ),
  );
}
